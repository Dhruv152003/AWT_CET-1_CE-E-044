                        function calculateBMI() 
                        {
                            let height = parseInt
                            (document.querySelector
                            ("#height").value);
                            let weight = parseInt
                            (document.querySelector
                            ("#weight").value);let result 
                            = document.querySelector
                            ("#result");
                            let bmi = (weight / 
                            ((height * height)/ 10000)).toFixed(2);
                            result.innerHTML = `${bmi}`;
                        }
